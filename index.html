Snippet #01: Debounce
function debounce<T>(fn: T, delay: number) {
  let timeoutId: any = null;
  return (...args: any[]) => {
    if (timeoutId) clearTimeout(timeoutId);
    timeoutId = setTimeout(() => fn(...args), delay);
  };
}
